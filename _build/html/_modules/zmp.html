[docs]class DIIS:
  def __init__(self, S, diis_space):
    eig,Z=np.linalg.eigh(S)
    S12=1./np.sqrt(eig)
    self.S=S
    self.O=reduce(np.dot,(Z,np.diag(S12),Z.T))
    self.diis_space=diis_space
    self.norb=len(S[0])
    self.ems=np.zeros((self.diis_space, self.norb, self.norb))
    self.pms=np.zeros((self.diis_space, self.norb, self.norb))


[docs]  def get_fock(self, iter, fock, dm ):
    if self.diis_space < 2:
      return fock, 0.0
    if iter <= 1 :
      return fock, 0.0

    for k in reversed(range(1, min(iter, self.diis_space))):
      self.ems[k] = self.ems[k-1][:]
      self.pms[k] = self.pms[k-1][:]

    em=reduce(np.dot, (fock, dm, self.S))
    em -=em.T

    self.ems[0]=reduce(np.dot, (self.O.T, em, self.O))
    self.pms[0]=fock[:]
    idx=np.abs(self.ems[0]).argmax()
    diis_error=np.abs(np.ravel(self.ems[0])[idx])

    #Solve BC=A to find C
    nb=min(iter, self.diis_space)-1
    B=-1.*np.ones((nb+1,nb+1))
    B[nb,nb]=0
    B[:nb,:nb]=np.einsum('aij,bji->ab',self.ems[:nb,:,:],self.ems[:nb,:,:])
    A=np.zeros(nb+1)
    A[nb]=-1
    C=np.linalg.solve(B, A)

    # form new interpolated diis fock matrix
    newfock = np.zeros_like(fock)
    for i,c in enumerate(C[:-1]):
      newfock += c*self.pms[i]

    return newfock, diis_error

[docs]  def initialize(self):
    self.h1e=self.get_hcore(self.mol)
    self.s1e=self.get_ovlp(self.mol)
    self.J_tar=scf.hf.get_jk(self.mol,self.dm_tar)[0]
    if self.guide is None:
      self.V0=np.zeros_like(self.dm_tar)
    elif self.guide.lower()=='fa':
      self.V0=((self.Ne-1.)/self.Ne)*(self.J_tar)
    else :
      self.mf.xc=self.guide
      self.V0=self.mf.get_veff(self.mol,dm=self.dm_tar)
    self.F0=self.h1e+self.V0
    self.initialized=True



[docs]  def zscf(self, l):
    if not self.initialized:
      self.initialize()
    self.l=l

    self.converged=False
    #SCF under given lambda l
    mol=self.mol
    self.zdiis=DIIS(self.s1e,self.diis_space)

    for cycle in range(1,self.max_cycle):
      self.J=scf.hf.get_jk(mol,self.dm)[0]

      self.F=self.F0+l*(self.J-self.J_tar)
      self.F=scf.hf.level_shift(self.s1e, self.dm*.5, self.F, self.level_shift_factor)
      self.F,diis_e=self.zdiis.get_fock(cycle, self.F, self.dm) #DIIS

      self.mo_energy,self.mo_coeff=scf.hf.eig(self.F,self.s1e)
      self.mo_occ=self.get_occ(self.mo_energy, self.mo_coeff)
      self.dm=self.make_rdm1(self.mo_coeff, self.mo_occ)
      #self.dm=2.*np.einsum('pi,qi->pq',self.mo_coeff[:,:self.Nocc], self.mo_coeff[:,:self.Nocc])
      #self.dm=lib.tag_array(self.dm, mo_coeff = self.mo_coeff, mo_occ=occ)

      ddm=self.dm_old-self.dm
      dm_e=np.max(np.abs(ddm))
      self.dm_old=self.dm
      dm_converged=dm_e < self.conv_tol_dm
      diis_converged=diis_e < self.conv_tol_diis 
      self.mo_energy[self.mo_occ==0]-=self.level_shift_factor
      print(self.mo_energy[3:7])

      gap=self.mo_energy[self.Nocc]-self.mo_energy[self.Nocc-1] #-self.shift*2
      print(f'\rlambda= {l:7.2f}  iter: {cycle:4d} gap= {gap:10.7f}   ',end='\r')

      self.converged = dm_converged and diis_converged
      if self.converged and cycle > 1:
        break

    self.J=scf.hf.get_jk(mol,self.dm)[0]
    dn=dft.numint.eval_rho(self.mol, self.ao, self.dm-self.dm_tar)
    dN=1000*np.einsum('r,r',abs(dn),self.weights)
    C=np.einsum('ij,ji',self.dm-self.dm_tar,self.J-self.J_tar) #Objective of minimization
    print(f'lambda= {l:7.2f} niter: {cycle:4d} gap= {gap:10.7f} dN= {dN:7.2f} C= {C:.2e} ')

    if not self.initialized:
      self.initialize()
    self.l=l

    self.converged=False
    #SCF under given lambda l
    mol=self.mol
    self.zdiis_a=DIIS(self.s1e,self.diis_space)
    self.zdiis_b=DIIS(self.s1e,self.diis_space)

    for cycle in range(1,self.max_cycle):
      self.J=scf.hf.get_jk(mol,self.dm)[0]

      self.Fa=self.F0[0]+2*l*(self.J[0]-self.J_tar[0])
      self.Fb=self.F0[1]+2*l*(self.J[1]-self.J_tar[1])

      self.Fa=scf.hf.level_shift(self.s1e, self.dm[0], self.Fa, self.level_shift_factor)
      self.Fb=scf.hf.level_shift(self.s1e, self.dm[1], self.Fb, self.level_shift_factor)

      self.Fa,diis_e_a=self.zdiis_a.get_fock(cycle, self.Fa, self.dm[0])
      self.Fb,diis_e_b=self.zdiis_b.get_fock(cycle, self.Fb, self.dm[1])

      e_a,c_a=scf.hf.eig(self.Fa,self.s1e)
      e_b,c_b=scf.hf.eig(self.Fb,self.s1e)
      self.mo_energy=np.array((e_a,e_b))
      self.mo_coeff=np.array((c_a,c_b))

      self.mo_occ=self.get_occ(self.mo_energy, self.mo_coeff)
      self.dm=self.make_rdm1(self.mo_coeff, self.mo_occ)
      #self.dm=2.*np.einsum('pi,qi->pq',self.mo_coeff[:,:self.Nocc], self.mo_coeff[:,:self.Nocc])
      #self.dm=lib.tag_array(self.dm, mo_coeff = self.mo_coeff, mo_occ=occ)

      ddm=self.dm_old-self.dm
      dm_e=np.max(np.abs(ddm))
      self.dm_old=self.dm
      dm_converged=dm_e < self.conv_tol_dm
      diis_converged=diis_e_a+diis_e_b < self.conv_tol_diis
      self.mo_energy[0][self.mo_occ[0]==0]-=self.level_shift_factor
      self.mo_energy[1][self.mo_occ[1]==0]-=self.level_shift_factor

      HOMO=np.maximum(self.mo_energy[0][self.nelec[0]-1],
                      self.mo_energy[1][self.nelec[1]-1])

                      self.mo_energy[1][self.nelec[1]])

      print(f'\rlambda= {l:7.2f}  iter: {cycle:4d} gap= {gap:10.7f}   ',end='\r')

      self.converged = dm_converged and diis_converged
      if self.converged and cycle > 1:
        break

    self.J=scf.hf.get_jk(mol,self.dm)[0]
