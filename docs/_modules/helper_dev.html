.. topic:: functions

    helper.Cart_Spharm(xyz, l)
        Calculate Cartesian spherical harmonics of given coordinates xyz
        for given angular momentum QN l
        Return : Z(ncoord, 2*l+1)

    helper.grid_refine(n_rad, n_ang, grid)
        Return : refined grid(radial order)

    helper.convert_coeff(c, x)
        Convert CubicSpline(x, y).c to ax^2+bx^2... formant
        Return : spline coefficient centerd at x = 0

    helper.eval_I(mol, P, Lvl, lmax)
        Calculate I for helper.eval_hartree function
        Return : I, radial grid, number of radial grid for each atom

    helper.eval_hartree(mol, P, Lvl, lmax, coords)
        For given density matrix P, calculate real-space Hartree potential at "coords"
        calls eval_I function
        Return : vH(coords)

    helper.eval_vxc_libxc(mol, P, func, coords)
        Calculate vxc(r) by numerical derivative
        Return : vxc_alpha, vxc_beta when P is UKS-rdm, vxc when P is RKS-rdm

[docs]def Cart_Spharm(xyz, lmax):
    """
    #Calculate Cartesian spherical harmonics Z for given xyz coordinate
    #from m = -l to l, return array Z of shape (ncoord, lmax+1, 2lmax+1)
    #for specific l & m, call Z[:, l, lmax+m]
    """
    ncoord = np.size(xyz[:, 0])
    rho = np.zeros((ncoord)) #distance from origin
    azi = np.zeros((ncoord)) #azimuth angle, theta in scipy
    pol = np.zeros((ncoord)) #polar angle, phi in scipy
    Z = np.zeros((ncoord, lmax+1, 2*lmax+1))     #Real spherical harmonics

    xy = xyz[:, 0]**2 + xyz[:, 1]**2
    rho = np.sqrt(xy + xyz[:, 2]**2)
    azi = np.arctan2(xyz[:, 1], xyz[:, 0])
    pol = np.arctan2(np.sqrt(xy), xyz[:, 2])

    a = np.sqrt(0.5)
    for l in range(lmax+1):
        for m in range(1, l+1):
            Yp = sph_harm(m, l, azi, pol)
            Yn = sph_harm(-m, l, azi, pol)
            Z[:, l, lmax-m] = a*np.real(1.j*(Yn-((-1.)**m)*Yp))
            Z[:, l, lmax+m] = a*np.real((Yn+((-1.)**m)*Yp))

        Z[:, l, lmax] = np.real(sph_harm(0, l, azi, pol))

    return Z



[docs]def grid_refine(n_rad, n_ang, grid):
    """Summary
        This function reshapes the radial grid outputs from pysef into an easier to manipulate form usef for KS inversion DC DFT comparison.

    Details
        pyscf returns 12-radi grid as groups if prune = None when generate DFT grids which are challenging to work with.
            This function converts that to all-radi grids as a group
            so that first n_ang grid belongs to same radial gridpoint

    Args:
        n_rad (?): radians?
        n_ang (?): Angles
        grid (?): Grid output from pyscf

    Returns:
        (numpy array): The grid points reshaped into ____.

    """
    nrest = n_rad%12
    nmain = (n_rad-nrest)//12

    m = grid[:nmain*12*n_ang]
    m = m.reshape(nmain, n_ang, 12)
    mmain = np.zeros((nmain*12, n_ang))
    for i in range(nmain):
        mmain[i*12:(i+1)*12, :] = m[i, :, :].T

    m = grid[nmain*12*n_ang:]
    mrest = m.reshape(n_ang, nrest).T
    return np.concatenate((mmain, mrest), axis=0).reshape(n_rad*n_ang)



[docs]def convert_coeff(c, x):
    """Summary
        CubicSpline(x, y).c gives coefficients corresponding to :math:`a*(x-xi)^3+b*(x-xi)^2`.
        This function returns coefficients corresponding to :math:`ax^2+bx^2`

    Args:
        c (?): ?
        x (?): ?

    Returns:
        (array ?): Description

    """
    nc = np.zeros_like(c)
    nx = np.size(x)
    for i in range(nx-1):
        nc[0, :] = c[0, :]
        nc[1, :] = c[1, :]-3*c[0, :]*x[:nx-1]
        nc[2, :] = c[2, :]-2*c[1, :]*x[:nx-1]+3*c[0, :]*x[:nx-1]**2
        nc[3, :] = c[3, :]-1*c[2, :]*x[:nx-1]+1*c[1, :]*x[:nx-1]**2-1*c[0, :]*x[:nx-1]**3
    return nc



[docs]def eval_I(mol, P, Lvl, lmax):
    """Summary


    Calculate I_ilm(ri) using eq(15) of Mirko et al. [Mirko2014]_
    Use this with eval_hartree

    .. math::

            I^{i}_{lm} (r_{i}) = \\frac{1}{r^{l+1}_{i}} \\int_{0}^{r_{i}} r^{\\prime l + 2} s^{i}_{lm} ( r^{\\prime} ) d r^{\\prime} + ...


    :References:

        .. [Mirko2014] Mirko Franchini, Pierre Herman Theodoor Philipsen, Erik van Lenthe, and Lucas Visscher.
                Accurate Coulomb Potentials for Periodic and Molecular Systems through Density Fitting. (2014)
                <https://doi.org/10.1021/ct500172n> Journal of Chemical Theory and Computation, 10(5), 1994-2004.

    Args:
        mol (?): ? a mol object?
        P (?): ?
        Lvl (?): ?
        lmax (?): ?

    Returns:
        (array ?): Description Itot
        (array ?): Description rlist
        (array ?): Description nr

    """
    nr = np.zeros(mol.natm)
    for ia in range(mol.natm):
        symb = mol.atom_symbol(ia)
        chg = gto.charge(symb)
        nr[ia] = int(dft.gen_grid._default_rad(chg, Lvl))
    maxnr = np.max(nr)
    Itot = np.zeros((mol.natm, int(maxnr), lmax+1, 2*lmax+1))
    rlist = np.zeros((mol.natm, int(maxnr)))

    for ia in range(mol.natm):
        symb = mol.atom_symbol(ia)
        chg = gto.charge(symb)
        n_rad = dft.gen_grid._default_rad(chg, Lvl)
        n_ang = dft.gen_grid._default_ang(chg, Lvl)
        rad, dr = dft.radi.gauss_chebyshev(n_rad, chg)
        grids = dft.gen_grid.gen_atomic_grids(mol, prune=None, level=Lvl)
        c = np.array(grids[symb][0]) #xyz coordinate centred at symb atom
        cx = grid_refine(n_rad, n_ang, c[:, 0])
        cy = grid_refine(n_rad, n_ang, c[:, 1])
        cz = grid_refine(n_rad, n_ang, c[:, 2])
        c = np.vstack((cx, cy, cz)).T
        r = np.sqrt(cx**2+cy**2+cz**2)
        w = np.array(grids[symb][1]) #weights
        w = grid_refine(n_rad, n_ang, w)
        Wrad = rad**2.*dr    #Radial weights
        sample_r = int(n_rad/2)
        Wang = w[sample_r*n_ang:(sample_r+1)*n_ang]/Wrad[sample_r] #Angular weights

        acoord = mol.atom_coord(ia)
        ao = numint.eval_ao(mol, c+acoord) #AO value in real coordinate
        if np.array(P).ndim == 2:
            rho_org = numint.eval_rho(mol, ao, P) #Rho in real coordinate
        elif np.array(P).ndim == 3:
            rho_org = numint.eval_rho(mol, ao, P[0]+P[1]) #Rho in real coordinate

        Acoord = mol.atom_coords()
        tst = c+acoord
        d = distance_matrix(Acoord, tst)
        p = np.exp(-2.*d)/(d**3)
        for ib in range(mol.natm):
            symb2 = mol.atom_symbol(ib)
            if symb2 == "H" or symb2 == "h":
                p[ib, :] = p[ib, :]*0.3

        rho_here = p[ia, :]/np.sum(p, axis=0)*rho_org

        Z = Cart_Spharm(c[sample_r*n_ang:(sample_r+1)*n_ang], lmax)
        w = w.reshape(n_rad, n_ang)
        rho_here = rho_here.reshape(n_rad, n_ang)
        C = np.einsum('alm,ra,a->rlm', Z, rho_here, Wang)

        #rho = np.einsum('rlm,alm->ra', C, Z) #evalute density from C, = rho_here
        #print("Nelec = %f"%np.einsum('ra,ra', rho, w)) # = nelectron correspondint to this atom

        I = np.zeros((n_rad, lmax+1, 2*lmax+1))
        vH = np.zeros((n_rad, n_ang))

        for l in range(0, lmax+1):
            for m in range(-l, l+1):
                f = CubicSpline(rad, C[:, l, lmax+m])
                c = f.c
                c = convert_coeff(c, rad)

                for i in range(0, n_rad):
                    I1 = 0.
                    I2 = 0.
                    for j in range(0, i):
                        for k in range(0, 4):
                            I1 = I1+c[k, j]/(6+l-k)*(rad[j+1]**(6+l-k)-rad[j]**(6+l-k))

                    for j in range(i, len(rad)-1):
                        if l < 2 or l > 5:
                            for k in range(0, 4):
                                I2 = I2+c[k, j]/(5-l-k)*(rad[j+1]**(5-l-k)-rad[j]**(5-l-k))
                        elif l == 2:
                            I2 = I2+c[3, j]*np.log(rad[j+1]/rad[j])
                            for k in range(0, 3):
                                I2 = I2+c[k, j]/(3-k)*(rad[j+1]**(3-k)-rad[j]**(3-k))
                        elif l == 3:
                            I2 = I2+.5*c[0, j]*(rad[j+1]**2-rad[j]**2)+c[1, j]*(rad[j+1]-rad[j])+c[2, j]*np.log(rad[j+1]/rad[j])-c[3, j]*(1./rad[j+1]-1./rad[j])
                        elif l == 4:
                            I2 = I2+c[0, j]*(rad[j+1]-rad[j])+c[1, j]*np.log(rad[j+1]/rad[j])-c[2, j]*(1./rad[j+1]-1./rad[j])-.5*c[3, j]*(1./rad[j+1]**2-1./rad[j]**2)
                        elif l == 5:
                            I2 = I2+c[0, j]*np.log(rad[j+1]/rad[j])
                            for k in range(1, 4):
                                I2 = I2-c[k, j]/k*(rad[j+1]**(-k)-rad[j]**(-k))
                    #for all j s.t. 0<j<i & i+1<j<maxr

                    I[i, l, lmax+m] = I1/(rad[i]**(l+1))+(rad[i]**l)*I2
                    vH[i, :] = vH[i, :]+4*np.pi/(2.*l+1.)*Z[:, l, lmax+m]*I[i, l, lmax+m] #if atom, this is total vH

                    Itot[ia, i, l, lmax+m] = I[i, l, lmax+m]
                    rlist[ia, :n_rad] = rad

                #for all radial index i ; 0<i<maxr
            #for all m
        #for all l
    #for all ai(atom index)
    return Itot, rlist, nr



[docs]def eval_hartree(mol, P, Lvl, lmax, coords):
    """Calculate Hartree potential for given coords
    Code reference : J. Chem. Theory Comput. 2014, 10, 1994−2004
    Lvl is pyscf radial/angular grid generation level, 1~9
    lmax is maximum angular momentum, l> = 2 recommended
    For large lmax, calculation may be slow
    """
    Itot, rlist, nr = eval_I(mol, P, Lvl, lmax)
    Acoord = mol.atom_coords()
    d = distance_matrix(Acoord, coords)
    vH = np.zeros((np.size(coords)//3))
    for ia in range(mol.natm):
        n = int(nr[ia])
        rel_coord = coords-Acoord[ia, :]
        Z = Cart_Spharm(rel_coord, lmax)
        for l in range(0, lmax+1):
            for m in range(-l, l+1):
                f = CubicSpline(rlist[ia, :n], Itot[ia, :n, l, lmax+m])
                v = f(d[ia, :])
                rel_coord = coords-Acoord[ia, :]
                vH[:] = vH[:] + 4.*np.pi/(2.*l+1)*Z[:, l, lmax+m]*v[:]
    return vH



[docs]def eval_vxc_libxc(mol, P, func, coords, delta=1e-7):
    """vxc(r) of given functional and density matrix
    """
    Ncoord = np.size(coords)//3
    ao = numint.eval_ao(mol, coords, deriv=1)

    def numderiv(aux, delta):
        nabla_res = np.zeros((3, Ncoord))
        nabla_res[0, :] = (aux[0, :]-aux[1, :])/(2.*delta)
        nabla_res[1, :] = (aux[2, :]-aux[3, :])/(2.*delta)
        nabla_res[2, :] = (aux[4, :]-aux[5, :])/(2.*delta)
        return nabla_res

    auxcoords = np.zeros((6, Ncoord, 3))
    for i in range(6):
        auxcoords[i, :, :] = coords[:, :]
        auxcoords[i, :, (i//2)] = coords[:, (i//2)]+delta*(-1.)**i

    if mol.spin == 0: #spin-unpolarized case
        mP = spin_tr(P)

        den = numint.eval_rho(mol, ao, mP, xctype='GGA')
        exc, vxc = dft.libxc.eval_xc(func, den, spin=mol.spin, deriv=1)[:2]

        auxvsigma = np.zeros((6, Ncoord))
        for i in range(6):
            auxao = numint.eval_ao(mol, auxcoords[i, :, :], deriv=1)
            auxden = numint.eval_rho(mol, auxao, mP, xctype='GGA')
            vxc = dft.libxc.eval_xc(func, auxden, spin=mol.spin, deriv=1)[1]
            auxvsigma[i, :] = vxc[1]

        ao = numint.eval_ao(mol, coords, deriv=2)
        den = numint.eval_rho(mol, ao, mP, xctype='mGGA')
        nabla_vsigma = numderiv(auxvsigma, delta)
        vxc = vxc[0]-2*(den[4, :]*vxc[1]+np.einsum('ir,ir->r', den[1:4, :], nabla_vsigma[:, :]))
        if np.array(P).ndim == 2: #RHF scheme
            return np.array(vxc)
        elif np.array(P).ndim == 3: #UHF scheme
            return np.array(vxc), np.array(vxc)
        else:
                print("!ERROR!!! SOMETHING IS WRONG SN RJM LOOKS AT THIS")


    elif mol.spin != 0: #spin-polarized case
        den_a = numint.eval_rho(mol, ao, P[0], xctype='GGA')
        den_b = numint.eval_rho(mol, ao, P[1], xctype='GGA')
        exc, vxc = dft.libxc.eval_xc(func, (den_a, den_b), spin=mol.spin, deriv=1)[:2]

        auxvsigma_aa = np.zeros((6, Ncoord))
        auxvsigma_ab = np.zeros((6, Ncoord))
        auxvsigma_bb = np.zeros((6, Ncoord))
        for i in range(6):
            auxao = numint.eval_ao(mol, auxcoords[i, :, :], deriv=1)
            auxden_a = numint.eval_rho(mol, auxao, P[0], xctype='GGA')
            auxden_b = numint.eval_rho(mol, auxao, P[1], xctype='GGA')
            vxc = dft.libxc.eval_xc(func, (auxden_a, auxden_b), spin=mol.spin, deriv=1)[1]
            auxvsigma_aa[i, :] = vxc[1][:, 0]
            auxvsigma_ab[i, :] = vxc[1][:, 1]
            auxvsigma_bb[i, :] = vxc[1][:, 2]

        nabla_vsigma_aa = numderiv(auxvsigma_aa, delta)
        nabla_vsigma_ab = numderiv(auxvsigma_ab, delta)
        nabla_vsigma_bb = numderiv(auxvsigma_bb, delta)

        ao = numint.eval_ao(mol, coords, deriv=2)
        den_a = numint.eval_rho(mol, ao, P[0], xctype='mGGA')
        den_b = numint.eval_rho(mol, ao, P[1], xctype='mGGA')

        vxc_a = vxc[0][:, 0]\
        -2.*(den_a[4, :]*vxc[1][:, 0]+np.einsum('ir,ir->r', den_a[1:4, :], nabla_vsigma_aa[:, :]))\
        -np.einsum('ir,ir->r', nabla_vsigma_ab, den_b[1:4, :])-den_b[4, :]*vxc[1][:, 1]

        vxc_b = vxc[0][:, 1]\
        -2.*(den_b[4, :]*vxc[1][:, 2]+np.einsum('ir,ir->r', den_b[1:4, :], nabla_vsigma_bb[:, :]))\
        -np.einsum('ir,ir->r', nabla_vsigma_ab, den_a[1:4, :])-den_a[4, :]*vxc[1][:, 1]

        return np.array(vxc_a), np.array(vxc_b)
