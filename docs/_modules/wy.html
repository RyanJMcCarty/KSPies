[docs]def get_Sijk(mol,pbas):
  mol2=mol.copy()
  mol2.basis=pbas
  mol2.build()
  grids=dft.gen_grid.Grids(mol)
  grids.level=5
  grids.build()
  coords=grids.coords
  weights=grids.weights
  ao1=dft.numint.eval_ao(mol,coords,deriv=0)
  ao2=dft.numint.eval_ao(mol2,coords,deriv=0)
  t0=time.time()
  #Sijk=np.einsum('ri,rj,rk,r->ijk,ao1,ao1,ao2,weights)
  Sijk=pf.ovlp_aab(weights,ao1,ao2,mol.nao_nr(),mol2.nao_nr(),np.size(weights))
  n1=mol.nao_nr()
  n2=mol2.nao_nr()
  t=(time.time()-t0)/60.
  print(f'Sijk calculation for n1:{n1} n2:{n2} basis functions: {t:.1f} min')
  return Sijk

[docs]def run(self):
  #Minimize objective function -W
  self.initialize()
  t=time.time()
  self.res = minimize(self.eval_Ws, x0=self.b, method=self.method,jac=self.eval_Gd,
                   tol=self.tol,options={'disp': False})
  self.b=self.res.x
  self.converged=self.res.success
  self.t_opt += time.time()-t



[docs]def wyscf(self):
  #Run WY until its self-consistent. This is required for hybrid guiding potentials
  self.totnit=0
  self.run()
  self.totnit += self.res.nit
  for i in range(40):
    self.scfcycle=i+1
    P_old=self.P
    self.P_aux=P_old
    self.run()
    self.totnit += self.res.nit
    self.dP=np.max(np.abs(np.array(P_old)-np.array(self.P)))
    if self.dP < 1e-6 : 
      break

[docs]def basic(self,mol,pbas,Sijk):
  self.mol=mol
  self.S = mol.intor('int1e_ovlp_sph')
  self.T = mol.intor_symmetric('int1e_kin')
  self.V = mol.intor_symmetric('int1e_nuc')

  if pbas is None:
    pbas=mol.basis

  self.mol2=mol.copy()
  self.mol2.basis=pbas
  self.mol2.build()

  if Sijk is None:
    if gto.mole.is_same_mol(mol,self.mol2,tol=1e-9):
      self.Sijk=mol.intor('int3c1e')
      self.Tp=mol.intor('int1e_kin_sph')
    else:
      self.Sijk=get_Sijk(mol,pbas)
      self.Tp=self.mol2.intor('int1e_kin_sph')
  else: #Sijk is given
    if gto.mole.is_same_mol(mol,self.mol2,tol=1e-8):
      if np.size(Sijk[0,0,:]) == self.mol2.nao_nr():
        if np.allclose(Sijk,self.mol2.intor('int1e_kin_sph')):
          self.Sijk=Sijk
          self.Tp=self.mol2.intor('int1e_kin_sph')
        else: #same dim but different bas
          print("Warning! potential basis and Sijk may not be consistent")
          print("Overwriting Sijk with mol2.intor('int3c1e')") 
          self.Sijk=self.mol2.intor('int3c1e')
          self.Tp=self.mol2.intor('int1e_kin_sph')
          #raise AssertionError ("potential basis and Sijk is not consistent")
      else: #different dim
        raise AssertionError ("potential basis and Sijk is not consistent")
    else: #abs != pbs
      if np.size(Sijk[0,0,:]) == self.mol2.nao_nr():
        self.Sijk=Sijk
        self.Tp=self.mol2.intor('int1e_kin_sph')
      else:
        raise AssertionError ("potential basis corresponding to Sijk should be given")

  self.tol=1e-6
  self.method='BFGS'
  self.scfcycle=0
  return self

    self.internal_b=np.ones_like(self.b)

    if self.P_aux is None :
      self.P_aux=self.P_tar
    if self.guide.lower()=='fa':
      J_tar=scf.hf.get_jk(self.mol,self.P_aux)[0]
      self.V0=((self.Ne-1.)/self.Ne)*(J_tar)
    elif self.guide.lower()=='not':
      self.V0=np.zeros_like(self.P_tar)
    else :
      mydft=dft.RKS(self.mol)
      mydft.xc=self.guide
      self.V0=mydft.get_veff(self.mol,dm=self.P_aux)
    self.F0=self.T+self.V+self.V0

    if not np.allclose(b,self.internal_b,rtol=1e-10,atol=1e-10):
      t=time.time()
      self.internal_b=b.copy()
      F=self.F0+np.einsum('t,ijt->ij',b,self.Sijk)
      self.fock=F
      self.E,self.C=scf.hf.eig(F,self.S)
      self.P=2.*np.einsum('pi,qi->pq',self.C[:,:self.Nocc],self.C[:,:self.Nocc])
      self.t_eig += time.time()-t



[docs]  def eval_Ws(self,b):
    self.get_PEC(b)
    t=time.time()

    Ws = np.einsum('ij,ji',self.T,self.P)
    Ws+= np.einsum('ij,ji',(self.V+self.V0),(self.P-self.P_tar))
    #Gd=np.einsum('ij,ijt->t',(self.P-self.P_tar), self.Sijk)
    Gd = pf.einsum_ij_ijt_2t((self.P-self.P_tar),self.Sijk,self.nbas,self.npot)
    Ws+= np.einsum('t,t',b,Gd)
    self.Ws=Ws

    if self.reg > 0: #Regularization on
      reg = self.reg*2*np.einsum('s,st,t',b,self.Tp,b)
    else :
      reg=0.
    self.t_ws += time.time()-t
    return -(Ws-reg)


  def _eval_Gd(self,b):
    self.get_PEC(b)
    t=time.time()
    #self.Gd=np.einsum('jk,jkt->t',(self.P-self.P_tar),self.Sijk)
    self.Gd = pf.einsum_ij_ijt_2t((self.P-self.P_tar),self.Sijk,self.nbas,self.npot)
    if self.reg > 0: #Regularization on
      self.Gd -= 4*self.reg*np.einsum('st,t->s',self.Tp,b)
    self.t_gd += time.time()-t

[docs]  def eval_Ws(self,b):
    self.get_PEC(b)
    t=time.time()

    #Ga=np.einsum('ij,ijt->t',(self.P[0]-self.P_tar[0]), self.Sijk)
    #Gb=np.einsum('ij,ijt->t',(self.P[1]-self.P_tar[1]), self.Sijk)
    Ga = pf.einsum_ij_ijt_2t((self.P[0]-self.P_tar[0]),self.Sijk,self.nbas,self.npot)
    Gb = pf.einsum_ij_ijt_2t((self.P[1]-self.P_tar[1]),self.Sijk,self.nbas,self.npot)
    Ws = np.einsum('ij,ji',self.T,(self.P[0]+self.P[1]))
    Ws += np.einsum('ij,ji',self.V,(self.P[0]+self.P[1]-self.P_tar[0]-self.P_tar[1]))
    Ws += np.einsum('ij,ji',self.V0[0],(self.P[0]-self.P_tar[0]))
    Ws += np.einsum('ij,ji',self.V0[1],(self.P[1]-self.P_tar[1]))
    Ws += np.einsum('t,t',b[:self.npot],Ga)
    Ws += np.einsum('t,t',b[self.npot:],Gb)
    self.Ws=Ws

    if self.reg > 0: #Regularization on
      reg = self.reg*2*np.einsum('s,st,t',b[:self.npot],self.Tp,b[:self.npot])
      reg+= self.reg*2*np.einsum('s,st,t',b[self.npot:],self.Tp,b[self.npot:])
    else:
      reg=0.
    self.t_ws += time.time()-t
    return -(Ws-reg)


  def _eval_Gd(self,b):
    self.get_PEC(b)
    t=time.time()
    #self.Ga=np.einsum('jk,jkt->t',(self.P[0]-self.P_tar[0]), self.Sijk)
    #self.Gb=np.einsum('jk,jkt->t',(self.P[1]-self.P_tar[1]), self.Sijk)
    self.Ga = pf.einsum_ij_ijt_2t((self.P[0]-self.P_tar[0]),self.Sijk,self.nbas,self.npot)
    self.Gb = pf.einsum_ij_ijt_2t((self.P[1]-self.P_tar[1]),self.Sijk,self.nbas,self.npot)
    if self.reg > 0: #Regularization on
      self.Ga -= 4*self.reg*np.einsum('st,t->s',self.Tp,b[:self.npot])
      self.Gb -= 4*self.reg*np.einsum('st,t->s',self.Tp,b[self.npot:])
    self.Gd = np.concatenate((self.Ga,self.Gb),axis=None)
    self.t_gd += time.time()-t
